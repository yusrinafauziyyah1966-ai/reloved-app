// ALUR TRANSAKSI TAWAR-MENAWAR DAN CHECKOUT RE-VIBE
[Mulai] Pembeli berkunjung -> Cari pakaian di halaman Jelajah
 │
 ├── Klik tombol 'Tawar Harga' (Detail Produk)
 │    ├── Masukkan nominal penawaran harga
 │    ├── Logika Auto-Bargain mengevaluasi tawaran
 │    │    ├── < 70% : Otomatis Ditolak bot penjual
 │    │    ├── ≥ 90% : Otomatis Diterima bot penjual
 │    │    └── 70% - 89% : Status PENDING (Menunggu disetujui penjual secara manual)
 │
 └── Tawaran Diterima -> Masuk Halaman Checkout
      ├── Hitung Ongkir (RajaOngkir) + Biaya Jasa Rekber Pembeli (5% Proteksi)
      ├── Bayar Aman via Payment Gateway (QRIS/VA)
      ├── Dana ditahan di Rekening Bersama Re-Vibe (Status: Escrow Safe)
      ├── Penjual mengirimkan pakaian -> Mengunggah nomor resi pengiriman
      └── Pembeli klik Konfirmasi Terima -> Dana diteruskan ke saldo dompet penjual [Selesai]

Skema Relasi Database MySQL (Entity Relationship)

users (Entitas Utama) PRIMARY
🔑 id : BIGINT (PK)
   name : VARCHAR(150)
   email : VARCHAR(100) (Unique)
   password : VARCHAR(255)
   role : ENUM('buyer','seller','admin')
   co2_saved_kg : DECIMAL(8,2)
   water_saved_liter : INT
   balance : DECIMAL(12,2)
products (Entitas Pakaian) FOREIGN
🔑 id : BIGINT (PK)
🔗 seller_id : BIGINT (FK -> users.id)
🔗 category_id : INT (FK -> categories.id)
   name : VARCHAR(200)
   description : TEXT
   brand : VARCHAR(100)
   size : VARCHAR(10)
   condition_rate : TINYINT (1-10)
   price : DECIMAL(12,2)
   status : ENUM('available','booked','sold')
   images : JSON
negotiations (Relasi Nego) FOREIGN
🔑 id : BIGINT (PK)
🔗 buyer_id : BIGINT (FK -> users.id)
🔗 product_id : BIGINT (FK -> products.id)
   proposed_price : DECIMAL(12,2)
   counter_price : DECIMAL(12,2)
   status : ENUM('pending','accepted','rejected')
   created_at : TIMESTAMP
transactions (Relasi Rekber) FOREIGN
🔑 id : BIGINT (PK)
   order_number : VARCHAR(50) (Unique)
🔗 buyer_id : BIGINT (FK -> users.id)
🔗 seller_id : BIGINT (FK -> users.id)
🔗 product_id : BIGINT (FK -> products.id)
   subtotal : DECIMAL(12,2)
   protection_fee : DECIMAL(12,2) (5%)
   shipping_fee : DECIMAL(12,2)
   total_amount : DECIMAL(12,2)
   payment_status : ENUM('unpaid','paid','expired')
   order_status : ENUM('pending','shipped','completed')
categories (Kamus Kategori) LOOKUP
🔑 id : INT (PK)
   name : VARCHAR(100)
   slug : VARCHAR(100)
   icon : VARCHAR(100)
